Button.tsx — v1.0 
const Button = ({ variant, children }) => {
  const baseClasses = "px-6 py-2.5 rounded-md font-semibold";
  const styles = {
    primary: "bg-primary text-on-primary shadow-sm",
    secondary: "bg-surface-container-highest text-on-surface",
    ghost: "text-primary hover:bg-primary-container/20"
  };

  return (
    <button className={`${baseClasses} ${styles[variant]}`}>
      {children}
    </button>
  );
};
